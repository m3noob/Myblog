def folderPath = '/path/to/your/folder' // Replace with the actual path to your folder

def folder = Jenkins.instance.getItemByFullName(folderPath)

if (folder) {
    println "Folder: ${folder.fullName}"

    folder.items.each { job ->
        println "\tJob: ${job.fullName}"

        job.builds.each { build ->
            def result = build.result ?: 'RUNNING'
            println "\t\tBuild #${build.number}, Result: $result"

            def buildStages = build.getAction(jenkins.model.BuildStages)
            if (buildStages != null) {
                def stageNodes = buildStages.getStages()
                if (stageNodes != null) {
                    stageNodes.each { stageNode ->
                        println "\t\t\tStage: ${stageNode.name}, Status: ${stageNode.getStatus()}"
                    }
                }
            }
        }
    }
} else {
    println "Folder not found!"
}
